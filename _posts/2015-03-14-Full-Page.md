**Javascript**
=============
{% highlight javascript %}
<script type="text/javascript">
  $(function(){
    $('#fullpage').fullpage({
      navigation: true,
      css3: true,
      paddingTop: '62px',
      anchors: ['s1', 's2', 's3'],
      navigationTooltips: ['section1', 'section2', 'section3'],
      slidesNavigation: true,
      afterLoad : function(anchor, index){
        $('*[data-anchor='+ anchor +']').find('em').fadeToggle('slow');
      }
    });
  }); 
</script>
{% endhighlight %}
